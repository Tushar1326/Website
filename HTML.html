1. Preprocessor Directives

Preprocessor directives in C are instructions processed before the actual compilation. These include macros, file inclusions, conditional compilations, and other utilities that help make the code modular and easier to manage. 

#include <stdio.h>
#define PI 3.14

#if defined(PI)
    #define AREA(r) (PI * r * r)
#endif

int main() {
    float radius = 5.0;
    printf("Area = %.2f\n", AREA(radius));
    return 0;
}

Output: Area = 78.50

2. Tokens in C

Tokens are the smallest elements in a C program, including keywords, identifiers, constants, strings, symbols, and operators. 

#include <stdio.h>

int main() {
    int age = 21; // 'int' keyword, 'age' identifier, '21' constant
    printf("Age = %d\n", age); // String, operator '=' and ';' are special symbols
    return 0;
}

Output: Age = 21

3. Input/Output Functions

Formatted I/O: Use printf() and scanf() for formatted input and output.

#include <stdio.h>

int main() {
    int a;
    printf("Enter a number: ");
    scanf("%d", &a);
    printf("Value: %d\n", a);
    return 0;
}

Output: Enter a number: 10 Value: 10

Unformatted I/O: Use getchar(), putchar(), gets(), and puts().

#include <stdio.h>

int main() {
    char ch;
    printf("Enter a character: ");
    ch = getchar();
    printf("You entered: ");
    putchar(ch);
    printf("\n");

    char name[30];
    printf("Enter your name: ");
    gets(name);    // Note: gets() is unsafe, used here for demonstration only
    printf("Your name is: ");
    puts(name);

    return 0;
}

Output: Enter a character: A You entered: A Enter your name: John Your name is: John

4. switch-case and goto

The switch statement handles multiple conditions. goto is used to jump to labeled parts of code. 

#include <stdio.h>

int main() {
    int choice = 2;
    switch(choice) {
      case 1: printf("One\n"); break;
      case 2: printf("Two\n"); break;
      default: printf("Other\n");
    }

    goto skip;
    printf("This won't print.\n");
    skip:
    printf("Jumped here!\n");

    return 0;
}

Output: Two Jumped here!

5. Arrays in C

Arrays store multiple values of the same type. 1D arrays are linear, while 2D arrays represent matrices. 

#include <stdio.h>

int main() {
    int arr[5] = {1, 2, 3, 4, 5}; // 1D array
    int mat[2][2] = {{1, 2}, {3, 4}}; // 2D array

    printf("1D array elements: ");
    for(int i=0; i<5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    printf("2D array elements:\n");
    for(int i=0; i<2; i++) {
        for(int j=0; j<2; j++) {
            printf("%d ", mat[i][j]);
        }
        printf("\n");
    }

    return 0;
}

Output: 1D array elements: 1 2 3 4 5 2D array elements: 1 2 3 4

6. Array Operations

You can perform addition, subtraction, multiplication, and transpose operations on arrays. 

#include <stdio.h>

int main() {
    int a[2][2] = {{1,2},{3,4}};
    int b[2][2] = {{5,6},{7,8}};
    int sum[2][2], diff[2][2], mul[2][2], transpose[2][2];

    // Addition
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        sum[i][j] = a[i][j] + b[i][j];
      }
    }

    // Subtraction
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        diff[i][j] = a[i][j] - b[i][j];
      }
    }

    // Multiplication
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        mul[i][j] = 0;
        for(int k=0; k<2; k++) {
          mul[i][j] += a[i][k] * b[k][j];
        }
      }
    }

    // Transpose of a
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        transpose[i][j] = a[j][i];
      }
    }

    printf("Sum:\n");
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        printf("%d ", sum[i][j]);
      }
      printf("\n");
    }

    printf("Difference:\n");
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        printf("%d ", diff[i][j]);
      }
      printf("\n");
    }

    printf("Multiplication:\n");
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        printf("%d ", mul[i][j]);
      }
      printf("\n");
    }

    printf("Transpose of matrix a:\n");
    for(int i=0; i<2; i++) {
      for(int j=0; j<2; j++) {
        printf("%d ", transpose[i][j]);
      }
      printf("\n");
    }

    return 0;
}

Output: Sum: 6 8 10 12 Difference: -4 -4 -4 -4 Multiplication: 19 22 43 50 Transpose of matrix a: 1 3 2 4

7. Pattern Programs

C can be used to generate patterns like squares, rectangles, and pyramids using nested loops. 

#include <stdio.h>

int main() {
    // Right pyramid pattern
    int rows = 5;
    for(int i = 1; i <= rows; i++) {
      for(int j = 1; j <= i; j++) {
        printf("* ");
      }
      printf("\n");
    }
    return 0;
}

Output: * * * * * * * * * * * * * * *

8. String Handling Functions

These are useful for manipulating and comparing strings. 

#include <stdio.h>
#include <string.h>

int main() {
    char str1[20] = "hello";
    char str2[20];

    strcpy(str2, str1);        // Copy str1 to str2
    printf("Copied string: %s\n", str2);

    printf("Length of str1: %lu\n", strlen(str1));

    // strupr() and strlwr() are not standard C but available in some compilers like Turbo C
    // Here's a manual way to convert to uppercase:
    for(int i=0; str1[i]; i++) {
      if(str1[i] >= 'a' && str1[i] <= 'z') {
        str1[i] = str1[i] - 32;
      }
    }
    printf("Uppercase str1: %s\n", str1);

    // strcmp() compares strings
    if(strcmp(str1, "HELLO") == 0) {
      printf("Strings are equal\n");
    } else {
      printf("Strings are not equal\n");
    }

    return 0;
}

Output: Copied string: hello Length of str1: 5 Uppercase str1: HELLO Strings are equal
